Basic Usage

{% endlang %} {% lang ru %}

Использование

{% endlang %}

{% lang en %}To create a simple UI, extend the CubeScreen class.{% endlang %} {% lang ru %}Чтобы создать простой интерфейс, унаследуйте класс CubeScreen.{% endlang %}

MyScreen.java
public class MyScreen extends CubeScreen {
    @Override
    protected void init() {
        super.init();
        
        // x, y, width, height, label
        CButton btn = new CButton(10, 10, 100, 20, "Click Me");
        btn.setOnClick(() -> {
            System.out.println("Hello World!");
        });
        
        this.add(btn);
    }
}
